examples/media/config.js

'use strict';

/**
 * @submodule bundle.media
 */

/**
 * @caption Media player example configuration
 *
 * @readme
 * OSE is configured by an configuration object, `module.exports` in this case. Each property of this
 * object defines the configuration for one [OSE plugin].
 *
 * @class bundle.examples.media.config
 * @type module
 */


// Basic properties of OSE instance
exports.ose = {
  // Name of this OSE instance
  name: 'media.example.org'
};


// Enable general control package
exports['ose-control'] = {};

// Enable general media player package
exports['ose-media'] = {};

// Enable PulseAudio control package
exports['ose-pa'] = {};

// Enable VideoLAN control package
exports['ose-videolan'] = {};

// Enable local filesystem package
exports['ose-fs'] = {};

// Enable Icecast package
exports['ose-icecast'] = {};

// Enable Youtube package
exports['ose-youtube'] = {};

// Enable X.Org control package
exports['ose-xorg'] = {};


// Enable CLI interface
exports.cli = {
  id: 'ose/lib/cli',

  // CLI can run some commands:
  script: [
    'wait 10000',
    'space example.org',
    'shard media',
    'entry player',
    'info',
//    'detail',
  ],
};


// Enable HTTP server
exports.http = {
  id: 'ose/lib/http',
  port: 4431,
};


// Enable HTML5 frontend
exports['ose-bb'] = {

  // Define dashboard content
  dashboard: [
    {
      caption: 'Media player',
      data: {
        pagelet: 'detail',
        space: 'example.org',
        shard: 'media',
        entry: 'player',
      }
    },
    {
      caption: 'X.Org remote control',
      data: {
        pagelet: 'gesture',
        space: 'example.org',
        shard: 'control',
        entry: 'xorg',
      }
    },
  ],
};


// Definition of data structure, the space named "example.org" contains all your data
exports.space = {
  id: 'ose/lib/space',        // Module id
  name: 'example.org',        // Name of the space
  home: 'media.example.org',  // Home instance of the space, this
                              // instance is the home instance of the
                              // space.

  // The space is partitioned into shards:
  shards: [

    // Control shard
    {
      sid: 2,                // Shard id unique within the space
      scope: 'control',      // Scope the shard belongs to
      alias: 'control',      // Shard alias
      entries: initControl,  // Method initializing entries belonging
                             // to the shard, defined below
    },

    // Media shard
    {
      sid: 3,              // Shard id unique within the space
      scope: 'media',      // Scope the shard belongs to
      alias: 'media',      // Shard alias
      entries: initMedia,  // Method initializing entries belonging
                           // to the shard, defined below
    },

    // Access to local filesystem
    {
      sid: 4,              // Shard id unique within the space
      scope: 'fs',         // Scope the shard belongs to
      alias: 'fs',         // Shard alias
      homeInit: function(shard) {
        // Setup shared root directory
        shard.root = '/opt/media';
      }
    }
  ]
};


// Xiph.org data namespace
exports.xiph = {
  id: 'ose/lib/space',
  name: 'xiph',
  home: 'media.example.org',
  shards: [{
    sid: 1,
    scope: 'icecast',
    alias: 'icecast'
  }]
};


// Google data namespace
exports.google = {
  id: 'ose/lib/space',
  name: 'google',
  home: 'media.example.org',
  shards: [{
    sid: 1,
    scope: 'youtube',
    alias: 'youtube'
  }]
};

// "control" shard initialization method.
function initControl(shard) {

  // Create X.Org server control entry
  shard.entry('xorg', 'xorg', {name: 'X.Org server'});

  // Create volume control entry
  shard.entry('volume', 'paDbus', {name: 'PulseAudio'});

  // Create playback control entry
  shard.entry('playback', 'vlc', {name: 'VLC'});
}


// "media" shard initialization method.
function initMedia(shard) {

  // Create entry representing generic Media player
  shard.entry(
    'player',  // Entry id
    'player',  // Entry kind

    // Entry data
    {
      name: 'Media Player',  // Displayed name
      playback: {  // Identification of playback control entry
        shard: 'control',
        entry: 'playback',
      },
      volume: {  // Identification of volume control entry
        shard: 'control',
        entry: 'volume',
      },

      /*
      dvb: {  // Identification of DVB streamer control entry
        shard: 'control',
        entry: 'dvb'
      },
      */

    }
  );

  // Run following method after plugins initialization.
  shard.afterHome(function() {
    addMedia('stream', require('./streams'));

//    addMedia('dvb', require('ose-bundle/examples/player/dvb'));
  });


  function addMedia(name, data) {
    for (var key in data) {
      // Create new media item entry
      shard.entry(key, name, data[key]);
    }
  }

}
